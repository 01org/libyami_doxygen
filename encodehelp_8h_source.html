85 static bool process_cmdline(int argc, char *argv[])
86 {
87  char opt;
88  const struct option long_opts[] = {
89  {"help", no_argument, NULL, 'h' },
90  {"qp", required_argument, NULL, 0 },
91  {"rcmode", required_argument, NULL, 0 },
92  {NULL, no_argument, NULL, 0 }};
93  int option_index;
94 
95  if (argc < 2) {
96  fprintf(stderr, "can not encode without option, please type 'yamiencode -h' to help\n");
97  return false;
98  }
99 
100  while ((opt = getopt_long_only(argc, argv, "W:H:b:f:c:s:i:o:N:h:", long_opts,&option_index)) != -1)
101  {
102  switch (opt) {
103  case 'h':
104  case '?':
105  print_help (argv[0]);
106  return false;
107  case 'i':
108  inputFileName = optarg;
109  break;
110  case 'o':
111  outputFileName = optarg;
112  break;
113  case 'W':
114  videoWidth = atoi(optarg);
115  break;
116  case 'H':
117  videoHeight = atoi(optarg);
118  break;
119  case 'b':
120  bitRate = atoi(optarg) * 1024;//kbps to bps
121  break;
122  case 'f':
123  fps = atoi(optarg);
124  break;
125  case 'c':
126  codec = optarg;
127  break;
128  case 's':
129  if (strlen(optarg) == 4)
130  inputFourcc = VA_FOURCC(optarg[0], optarg[1], optarg[2], optarg[3]);
131  break;
132  case 'N':
133  frameCount = atoi(optarg);
134  break;
135  case 0:
136  switch (option_index) {
137  case 1:
138  initQp = atoi(optarg);
139  break;
140  case 2:
141  rcMode = string_to_rc_mode(optarg);
142  break;
143  }
144  }
145  }
146 
147  if (!inputFileName) {
148  fprintf(stderr, "can not encode without input file\n");
149  return false;
150  }
151 
152  if ((rcMode == RATE_CONTROL_CBR) && (bitRate <= 0)) {
153  fprintf(stderr, "please make sure bitrate is positive when CBR mode\n");
154  return false;
155  }
156 
157  if (!strncmp(inputFileName, "/dev/video", strlen("/dev/video")) && !frameCount)
158  frameCount = 50;
159 
160  if (!outputFileName)
161  outputFileName = "test.264";
162 
163  return true;
164 }
