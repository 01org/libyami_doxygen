186 {
187  bool ret;
188  PicturePtr picture;
189  AutoLock l(m_lock);
190  picture = std::tr1::dynamic_pointer_cast<VaapiEncPicture>(pic);
191  if (picture) {
192  m_output.push_back(picture);
193  ret = true;
194  } else {
